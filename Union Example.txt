 B.CODE_COMBINATION_ID
,B.PERIOD_NAME PERIOD
,SUM(NVL(B.BEGIN_BALANCE_DR,0)-NVL(B.BEGIN_BALANCE_CR,0)) BEG_BALANCE
FROM
GL.GL_BALANCES B
WHERE
B.CODE_COMBINATION_ID LIKE '9980'
AND B.PERIOD_NAME LIKE 'JAN-00'
