SELECT ALL 
substr(APPLSYS.FND_REQUEST_GROUPS.REQUEST_GROUP_NAME, 0, 30) RQ_GROUP_NAME, 
substr(APPLSYS.FND_CONCURRENT_PROGRAMS.CONCURRENT_PROGRAM_NAME, 0, 35) CCM_PROG_NAME, 
substr(APPLSYS.FND_CONCURRENT_PROGRAMS_TL.USER_CONCURRENT_PROGRAM_NAME, 0, 70) USER_CCM_PROG_NAME, 
substr(APPLSYS.FND_APPLICATION_TL.APPLICATION_NAME, 0, 30) APPL_NAME, 
substr(APPLSYS.FND_CONCURRENT_PROGRAMS.ENABLED_FLAG, 0, 30) ENABLED_FLAG
FROM 
APPLSYS.FND_REQUEST_GROUP_UNITS, 
APPLSYS.FND_REQUEST_GROUPS, 
APPLSYS.FND_CONCURRENT_PROGRAMS, 
APPLSYS.FND_CONCURRENT_PROGRAMS_TL, 
APPLSYS.FND_APPLICATION_TL
WHERE 
APPLSYS.FND_REQUEST_GROUPS.REQUEST_GROUP_NAME LIKE '&enter_request_group_name'
 AND  ((APPLSYS.FND_REQUEST_GROUP_UNITS.REQUEST_GROUP_ID=APPLSYS.FND_REQUEST_GROUPS.REQUEST_GROUP_ID)
 AND (APPLSYS.FND_REQUEST_GROUP_UNITS.REQUEST_UNIT_ID=APPLSYS.FND_CONCURRENT_PROGRAMS.CONCURRENT_PROGRAM_ID)
 AND (APPLSYS.FND_CONCURRENT_PROGRAMS_TL.CONCURRENT_PROGRAM_ID=APPLSYS.FND_CONCURRENT_PROGRAMS.CONCURRENT_PROGRAM_ID)
 AND (APPLSYS.FND_APPLICATION_TL.APPLICATION_ID=APPLSYS.FND_CONCURRENT_PROGRAMS.APPLICATION_ID))
ORDER BY
APPLSYS.FND_CONCURRENT_PROGRAMS_TL.USER_CONCURRENT_PROGRAM_NAME ASC
/
