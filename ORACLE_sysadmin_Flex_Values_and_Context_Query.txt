SELECT ALL 
APPLSYS.FND_DESCR_FLEX_COL_USAGE_TL.DESCRIPTIVE_FLEX_CONTEXT_CODE,
APPLSYS.FND_FLEX_VALUE_SETS.FLEX_VALUE_SET_NAME, 
APPLSYS.FND_FLEX_VALUES.FLEX_VALUE_SET_ID,
APPLSYS.FND_FLEX_VALUES.FLEX_VALUE_ID, 
APPLSYS.FND_FLEX_VALUES.FLEX_VALUE,
APPLSYS.FND_DESCR_FLEX_CONTEXTS.DESCRIPTIVE_FLEX_CONTEXT_CODE,
APPLSYS.FND_FLEX_VALUES.ATTRIBUTE1, 
APPLSYS.FND_DESCRIPTIVE_FLEXS.TITLE, 
APPLSYS.FND_USER.USER_NAME,
APPLSYS.FND_DESCR_FLEX_CONTEXTS.LAST_UPDATE_DATE
FROM 
APPLSYS.FND_DESCR_FLEX_COL_USAGE_TL,
APPLSYS.FND_DESCRIPTIVE_FLEXS, 
APPLSYS.FND_FLEX_VALUES, 
APPLSYS.FND_FLEX_VALUE_SETS,
APPLSYS.FND_USER,
APPLSYS.FND_DESCR_FLEX_CONTEXTS
WHERE 
(APPLSYS.FND_FLEX_VALUE_SETS.FLEX_VALUE_SET_NAME LIKE '%KUEGL_X_STAT%'
AND APPLSYS.FND_DESCRIPTIVE_FLEXS.TITLE LIKE '%Flexfield Segment Values%'
AND APPLSYS.FND_FLEX_VALUES.FLEX_VALUE_ID=51859)
AND  (APPLSYS.FND_FLEX_VALUE_SETS.FLEX_VALUE_SET_ID=APPLSYS.FND_FLEX_VALUES.FLEX_VALUE_SET_ID
AND APPLSYS.FND_FLEX_VALUE_SETS.FLEX_VALUE_SET_NAME=APPLSYS.FND_DESCR_FLEX_CONTEXTS.DESCRIPTIVE_FLEX_CONTEXT_CODE
AND APPLSYS.FND_DESCR_FLEX_CONTEXTS.DESCRIPTIVE_FLEX_CONTEXT_CODE=APPLSYS.FND_DESCR_FLEX_COL_USAGE_TL.DESCRIPTIVE_FLEX_CONTEXT_CODE
AND APPLSYS.FND_USER.USER_ID=APPLSYS.FND_DESCR_FLEX_CONTEXTS.LAST_UPDATED_BY
AND APPLSYS.FND_DESCR_FLEX_CONTEXTS.DESCRIPTIVE_FLEXFIELD_NAME=APPLSYS.FND_DESCRIPTIVE_FLEXS.DESCRIPTIVE_FLEXFIELD_NAME);
