-- POINT to local TEMP table 
CREATE OR REPLACE PUBLIC SYNONYM EDS_AR_CUSTOMERS_INT_SYN
FOR APPS.TMP_EDS_AR_CUSTOMERS_INT;


-- POINT to RCOM tables using DB link
CREATE OR REPLACE PUBLIC SYNONYM EDS_AR_CUSTOMERS_INT_SYN
FOR FRC.EDS_AR_CUSTOMERS_INT@RCOM.WORLD;



UPDATE TMP_EDS_AR_CUSTOMERS_INT A
SET A.PROCESSED_FLAG = 'I'
WHERE A.PROCESSED_FLAG = 'H'
AND EXISTS (SELECT 1 FROM FRC.EDS_AR_CUSTOMERS_INT@RCOM.WORLD B
   WHERE B.CUSTOMER_NUMBER = A.CUSTOMER_NUMBER
   AND B.CUSTOMER_ADDRESS_ID = A.CUSTOMER_ADDRESS_ID);

COMMIT;
